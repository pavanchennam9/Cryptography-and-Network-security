// Main calculation function calculate() { let p = parseInt(document.getElementById("p").value); let g = parseInt(document.getElementById("g").value); let a = parseInt(document.getElementById("a").value); if (b === 0) { alert("Generate Bob's key first!"); return; } // Public keys let A = power(g, a, p); let B = power(g, b, p); // Shared keys let keyA = power(B, a, p); let keyB = power(A, b, p); // Display results document.getElementById("A").innerText = A; document.getElementById("B").innerText = B; document.getElementById("keyA").innerText = keyA; document.getElementById("keyB").innerText = keyB; }
